function toCamelCase(str){
  let arr = str.split('');
  let newArr = [];
  for(let i = 0; i < arr.length; i++) {
    if(arr[i - 1] == '_' || arr[i - 1] == '-') {
      newArr.push(arr[i].toUpperCase());
    } else {
      newArr.push(arr[i]);
    }
  }
  let newStr = newArr.join('').replace(/[-_]/g, '');
  return newStr;
}
